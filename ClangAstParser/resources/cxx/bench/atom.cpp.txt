class Atom {

   Atom(double r[3]) {
      this->clear();
      this->r[0] = r[0];
      this->r[1] = r[1];
      this->r[2] = r[2];
   }
   
   Atom(double r[3], basic_string<char> const *name) : stype("") {
      this->clear();
      this->r[0] = r[0];
      this->r[1] = r[1];
      this->r[2] = r[2];
      this->name = name;
   }
   
   void clear();
   
  
   
};
