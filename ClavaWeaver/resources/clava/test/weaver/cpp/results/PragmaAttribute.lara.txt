/**** File 'pragma_attribute.cpp' ****/


int foo(int a);

int foo2(int a);

int main() {
   int acc = 0;
   #pragma clava attribute initValue(int i=(10)) isParallel
   for(int i=(10); i < 10; i++) {
      #pragma clava attribute select(call) name(foo2)
      #pragma clava attribute select(call.arg)
      acc += foo2(20);
   }
}
