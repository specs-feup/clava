}


int fooNoDecl(int a) {
   
   return a + 10;
}

   (*a)++;
}

   wrap_foo(10);
   fooNoDecl(20);

/**** File 'clava_weaver_wrappers.cpp' ****/

#include "wrap.h"
#include <stdio.h>
#include "clava_weaver_wrappers.h"

int wrap_foo(int a) {
   int result;
   result = foo(a);
   
   return result;
}


void wrap_voidFoo(int * a) {
   voidFoo(a);
}


int wrap_printf(char const * __restrict__ A, ...) {
   int result;
   // Before call
   result = printf(A);
   
   return result;
}
